# PM-OS v0.1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a custom Electron desktop app that embeds Slack, Notion, and a web browser via WebContentsView, with an AI context bar, project management, and automation — the "Product Manager Operating System."

**Architecture:** Custom Electron app (not a literal VS Code fork for v0.1) with a plugin-based extension system. The main process manages WebContentsView instances for embedding third-party sites. A renderer process provides the shell UI (sidebar, tab bar, command palette, status bar). Extensions are TypeScript packages loaded at startup that register commands, panels, and sidebar views. Three shared packages (types, event-bus, claude) provide the cross-extension contract.

**Tech Stack:** Electron 35+, TypeScript 5.x, pnpm workspaces, esbuild, Vitest, Hono (auth backend), marked + mermaid (markdown), Anthropic SDK (AI)

```typescript
import type { Disposable, EventName, EventPayload, EventHandler } from '@pm-os/types';

export class EventBus {
  private listeners = new Map<string, Set<EventHandler<any>>>();

  on<E extends EventName>(event: E, handler: EventHandler<E>): Disposable {
    if (!this.listeners.has(event)) {
      this.listeners.set(event, new Set());
    }
    this.listeners.get(event)!.add(handler);
    return {
      dispose: () => { this.listeners.get(event)?.delete(handler); },
    };
  }

  emit<E extends EventName>(event: E, payload: EventPayload<E>): void {
    const handlers = this.listeners.get(event);
    if (handlers) {
      for (const handler of handlers) handler(payload);
    }
  }

  removeAllListeners(): void { this.listeners.clear(); }
}
```

- [ ] Implement `packages/claude/src/cache.ts`:

```typescript
interface CacheEntry { value: string; timestamp: number; }
interface SummaryCacheOptions { ttlMs?: number; maxSize?: number; }

export class SummaryCache {
  private entries = new Map<string, CacheEntry>();
  private ttlMs: number;
  private maxSize: number;

  constructor(options: SummaryCacheOptions = {}) {
    this.ttlMs = options.ttlMs ?? 5 * 60 * 1000;
    this.maxSize = options.maxSize ?? 100;
  }

  get(key: string): string | undefined {
    const entry = this.entries.get(key);
    if (!entry) return undefined;
    if (Date.now() - entry.timestamp > this.ttlMs) {
      this.entries.delete(key);
      return undefined;
    }
    return entry.value;
  }

  set(key: string, value: string): void {
    if (this.entries.size >= this.maxSize) {
```

```typescript
import type { AiCostEstimate } from '@pm-os/types';

const MODEL_PRICING: Record<string, { input: number; output: number }> = {
  'claude-sonnet-4-6': { input: 3.0, output: 15.0 },
  'claude-haiku-4-5': { input: 0.80, output: 4.0 },
  'claude-opus-4-6': { input: 15.0, output: 75.0 },
};

interface CostTrackerOptions { limitUsd?: number; }

export class CostTracker {
  private records: AiCostEstimate[] = [];
  private limitUsd: number | null;

  constructor(options: CostTrackerOptions = {}) {
    this.limitUsd = options.limitUsd ?? null;
  }

  record(estimate: AiCostEstimate): void { this.records.push(estimate); }

  get totalCostUsd(): number {
    return this.records.reduce((sum, r) => sum + r.estimatedCostUsd, 0);
  }
  get totalInputTokens(): number {
    return this.records.reduce((sum, r) => sum + r.inputTokens, 0);
  }
  get totalOutputTokens(): number {
    return this.records.reduce((sum, r) => sum + r.outputTokens, 0);
  }
  get isOverLimit(): boolean {
    if (this.limitUsd === null) return false;
    return this.totalCostUsd > this.limitUsd;
  }

  static estimate(inputTokens: number, outputTokens: number, model = 'claude-sonnet-4-6'): AiCostEstimate {
    const pricing = MODEL_PRICING[model] ?? MODEL_PRICING['claude-sonnet-4-6'];
    const cost = (inputTokens / 1_000_000) * pricing.input + (outputTokens / 1_000_000) * pricing.output;
    return { inputTokens, outputTokens, estimatedCostUsd: cost };
  }
}
```

```typescript
import type { ClaudeClientOptions } from '@pm-os/types';
import { SummaryCache } from './cache.js';
import { CostTracker } from './cost.js';

export class ClaudeClient {
  private anthropic: Anthropic;
  private model: string;
  private maxTokens: number;
  private cache: SummaryCache;
  readonly costTracker: CostTracker;

  constructor(options: ClaudeClientOptions) {
    this.anthropic = new Anthropic({ apiKey: options.apiKey });
    this.model = options.model ?? 'claude-sonnet-4-6';
    this.maxTokens = options.maxTokens ?? 4096;
    this.cache = new SummaryCache();
    this.costTracker = new CostTracker({ limitUsd: options.costLimitUsd });
  }

  async complete(systemPrompt: string, userMessage: string, options: { cacheKey?: string; maxTokens?: number } = {}): Promise<string> {
    if (options.cacheKey) {
      const cached = this.cache.get(options.cacheKey);
      if (cached) return cached;
    }
    if (this.costTracker.isOverLimit) {
      throw new Error('AI cost limit exceeded. Increase limit in project settings.');
    }

    const response = await this.anthropic.messages.create({
      model: this.model,
      max_tokens: options.maxTokens ?? this.maxTokens,
      system: systemPrompt,
      messages: [{ role: 'user', content: userMessage }],
    });

    const text = response.content
      .filter((block): block is Anthropic.TextBlock => block.type === 'text')
      .map((block) => block.text)
      .join('');

    const estimate = CostTracker.estimate(response.usage.input_tokens, response.usage.output_tokens, this.model);
    this.costTracker.record(estimate);

    if (options.cacheKey) this.cache.set(options.cacheKey, text);
    return text;
  }

  async stream(systemPrompt: string, userMessage: string, onChunk: (text: string) => void): Promise<string> {
    if (this.costTracker.isOverLimit) throw new Error('AI cost limit exceeded.');

    const stream = this.anthropic.messages.stream({
      model: this.model,
      max_tokens: this.maxTokens,
      system: systemPrompt,
      messages: [{ role: 'user', content: userMessage }],
    });

    let full = '';
    for await (const event of stream) {
      if (event.type === 'content_block_delta' && event.delta.type === 'text_delta') {
        full += event.delta.text;
        onChunk(event.delta.text);
      }
    }

    const finalMessage = await stream.finalMessage();
    const estimate = CostTracker.estimate(finalMessage.usage.input_tokens, finalMessage.usage.output_tokens, this.model);
    this.costTracker.record(estimate);
    return full;
  }
}
```

```typescript
import { BrowserWindow, WebContentsView } from 'electron';
import type { PanelBounds, WebContentsViewOptions, WebContentsViewState } from '@pm-os/types';

export class WcvManager {
  private views = new Map<string, WebContentsView>();
  private window: BrowserWindow;

  constructor(window: BrowserWindow) { this.window = window; }

  async create(options: WebContentsViewOptions): Promise<string> {
    const view = new WebContentsView({
      webPreferences: {
        partition: options.partition ?? ('persist:' + options.id),
        contextIsolation: true,
        nodeIntegration: false,
        sandbox: true,
      },
    });

    this.views.set(options.id, view);
    this.window.contentView.addChildView(view);

    if (options.bounds) view.setBounds(options.bounds);

    view.webContents.on('did-navigate', (_e, url) => {
      this.window.webContents.send('wcv:url-changed', { id: options.id, url });
    });
    view.webContents.on('did-navigate-in-page', (_e, url) => {
      this.window.webContents.send('wcv:url-changed', { id: options.id, url });
    });
    view.webContents.on('page-title-updated', (_e, title) => {
      this.window.webContents.send('wcv:title-changed', { id: options.id, title });
    });
    view.webContents.on('did-start-loading', () => {
      this.window.webContents.send('wcv:loading', { id: options.id, loading: true });
    });
    view.webContents.on('did-stop-loading', () => {
      this.window.webContents.send('wcv:loading', { id: options.id, loading: false });
    });

    await view.webContents.loadURL(options.url);
    return options.id;
  }

  async navigate(id: string, url: string): Promise<void> {
    const view = this.views.get(id);
    if (!view) throw new Error('WebContentsView not found: ' + id);
    await view.webContents.loadURL(url);
  }

  setBounds(id: string, bounds: PanelBounds): void {
    this.views.get(id)?.setBounds(bounds);
  }

  destroy(id: string): void {
    const view = this.views.get(id);
    if (!view) return;
    this.window.contentView.removeChildView(view);
    view.webContents.close();
    this.views.delete(id);
  }

  getState(id: string): WebContentsViewState | null {
    const view = this.views.get(id);
    if (!view) return null;
    return {
      id,
      url: view.webContents.getURL(),
      title: view.webContents.getTitle(),
      loading: view.webContents.isLoading(),
      canGoBack: view.webContents.canGoBack(),
      canGoForward: view.webContents.canGoForward(),
    };
  }

  goBack(id: string): void { this.views.get(id)?.webContents.goBack(); }
  goForward(id: string): void { this.views.get(id)?.webContents.goForward(); }
  reload(id: string): void { this.views.get(id)?.webContents.reload(); }
  destroyAll(): void { for (const id of this.views.keys()) this.destroy(id); }
}
```

```typescript
export function parseSlackUrl(url: string): SlackContext {
  try {
    const u = new URL(url);
    if (!u.hostname.endsWith('.slack.com') && u.hostname !== 'app.slack.com')
      return { workspace: null, channelId: null, threadTs: null };
    const workspace = u.hostname.replace('.slack.com', '');
    const parts = u.pathname.split('/').filter(Boolean);
    const clientIdx = parts.indexOf('client');
    const channelId = (clientIdx !== -1 && parts.length > clientIdx + 2) ? parts[clientIdx + 2] : null;
    const threadTs = u.searchParams.get('thread_ts') ?? null;
    return { workspace, channelId, threadTs };
  } catch { return { workspace: null, channelId: null, threadTs: null }; }
}
```
